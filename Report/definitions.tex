\begin{defn}
  \label{def:hdiv}
The divergent Sobolev spac is defined by
\[
  H(\div; \O) := \set{ \sigma \in [L^2(\O)]^2 : \div \sigma \in L^2(\O)}.
  \]
\end{defn}
