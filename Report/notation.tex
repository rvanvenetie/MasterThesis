\nomenclature[a1]{$\O$}{A bounded polyhedral domain in $\R^n$}
\nomenclature[a2]{$u$}{Almost always used to denote the \emph{exact} solution of the boundary-value problem at hand}
\nomenclature[b1]{$a \lesssim b$}{There holds $a \leq C_{st} b$ for some constant $C_{st}$}
\nomenclature[b2]{$a \gtrsim b$}{There holds $b \lesssim a$}
\nomenclature[b3]{$a \eqsim b$}{If $a \lesssim b$ and $a \gtrsim b$}
\nomenclature[c2]{$\v{v}$}{Used to indicate that $v$ is a vector-valued function}
\nomenclature[d0]{$C^m(\O)$}{The space of $m$-times continuously differentiable functions}
\nomenclature[d1]{$L^2(\O)$}{Space of square-integrable functions over $\O$}
\nomenclature[d2]{$H^m(\O)$}{Sobolev space of $L^2(\O)$ functions with weak derivatives up to order $m$}
\nomenclature[d3]{$H_0^m(\O)$}{Subspace of $H^m(\O)$ of all functions with vanishing trace on $\partial \O$}
\nomenclature[d4]{$H(\div; \O)$}{Space of vector-valued  $\left[L^2(\O)\right]^2$ functions with a weak divergence} 
\nomenclature[e01]{$\ip{f,g}_{\O}$}{The regular (vectur-valued) $L^2(\O)$ inner product}
\nomenclature[e1]{$\norm{\cdot}_{\O}$}{The regular (vector-valued) $L^2(\O)$ norm}
\nomenclature[e2]{$\norm{\cdot}_{H^m(\O)}$}{The Sobolev norm of order $m$ on $\O$}
\nomenclature[e3]{$\abs{\cdot}_{\O}$}{The Sobolev seminorm of order $m$ on $\O$}
\nomenclature[e4]{$\enorm{\cdot}_{\O}$}{The energy norm, i.e. $\enorm{v}_{\O}^2 = a(v,v)$}
\nomenclature[f1]{$\mathcal{T}$}{A conforming triangulation of $\O$}
\nomenclature[f2]{$\V, \E$}{The vertices and edges of a given triangulation $\mathcal{T} \subset \R^2$}
\nomenclature[f3]{$\w_a$}{The star patch centered at vertex $a \in \V$: all the triangles that touch $a$}
\nomenclature[g1]{$\P_p(K)$}{The space of polynomials of degree $p$ for an element $K$}
\nomenclature[g1]{$\P_p(\T)$}{$ := \set{f \in C(\O) : f|_{K} \in \P_p(K), \, K \in \T}$}
\nomenclature[g2]{$\P_p^{-1}(\T)$}{$ := \prod_{K \in \T} \P_p(K)$}
\nomenclature[g3]{$\RT_p(K)$}{$ := \left[\P_p(K)\right]^2 + \P_p(K)\v{x}$}
\nomenclature[g4]{$\RT^{-1}_p(\T)$}{$:= \prod_{K \in \T} \RT_p(K)$}
\nomenclature[g5]{$\RT^{-1}_{p,0}(\w_a)$}{$ := \set{ \v{\sigma} \in \RT_p^{-1}(\w_a) : \v{\sigma} \cdot \v{n_e} = 0\, e \in \partial \w_a \setminus \O}$}
\nomenclature[g6]{$\RT_p(\T)$}{$ := H(\div; \O) \cap \RT_p^{-1}(\T)$}
\nomenclature[g7]{$\RT_{p,0}(\w_a)$}{$ := H(\div; \O) \cap \RT_{p,0}^{-1}(\w_a)$}
\nomenclature[h1]{$\llbracket \vsig \rrbracket$}{ $\llbracket \vsig \rrbracket(x) := \lim_{\epsilon \to 0} \vsig (x + \epsilon n) \cdot n - \vsig(x - \epsilon n)\cdot n$, the jump over an interface in the direction of the unit normal $n$}
